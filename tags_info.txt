# show commit ids 
git log --abbrev-commit
git log --pretty=oneline --abbrev-commit

# push specific tag
git push origin <tagname>
eg: git push origin v1.5

# push all tags
git push origin --tags
git push origin -f --tags    # force updating

# delete specific tag in remote repo
git push origin --delete <tagname>

# delete a tag
git tag -d <tagname>

# show tag starting with v1.8. 
git tag -l 'v1.8.*'
